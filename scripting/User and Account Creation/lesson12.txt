#!/bin/bash

#display user ID adn user name of the user executing the script
#display if the user if the root user or not.

echo "Your UID is ${UID}"

USER_NAME=$(id -un)
echo "Your username is ${USER_NAME}"

#[[ is a bash specific cover for expressions
if [[ "${UID}" -eq 0 ]]
then

else 
	echo 'You are not root.'
fi
