!#/bin/bash


#####Script creates an account on the local system
#####User if pronted for their account name and password.



#####Ask for the user name.
read -p 'Enter the username to create: ' USERNAME

##### Ask for the real name.
read -p 'Enter the name of the person who this account is for: ' REALNAME

##### Ask for the password
read -p 'Enter a password for the account: ' PASSWORD


##### Create the user
dscl . -create -c "${PASSWORD}" -m " ${USERNAME}"

##### Set the password for the user.
echo ${PASSWORD} | PASSWD --stdin ${USERNAME}

##### Force password change on the first login
passwd -e ${USERNAME}
