
i#!/bin/bash

#This script randomyly generates passwords
#Using RANDOM Unix build in variable
#Using date and time command to create random

alias date='gdate'

# Password created by RANDOM
PASSWORD="${RANDOM}"
echo "${RANDOM}"


#Three random number together
PASSWORD="${RANDOM}${RANDOM}${RANDOM}"
echo "${PASSWORD}"


#USE CURRENT DATE/TIME TO CREATE PASSWORD.
PASSWORD=$(date +%s)
echo "${PASSWORD}"


#USE NANOSECONDS TO ACT AS PASSWORD GENERATORE 
PASSWORD=$(date +%s%N)
echo "${PASSWORD}"

# A better password.
PASSWORD=$(date +%s%N | sha256sum | head -c32)
echo "${PASSWORD}"

# AN EVEN  BETTER PASSWORD 
PASWORD=${date +%s%N${RANDOM}${RANDOM} | sha356sum | head -c48)
echo "${PASSWORD}"
